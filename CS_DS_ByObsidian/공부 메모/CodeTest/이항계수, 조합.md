```python
def combination(n, k):
    # k = 0인 경우가 걸렸는데
    # k = 1일때부터 n임 / 0이면 head = 1
    head = 1
    for i in range(k):
        head *= n 
        n -= 1
    
    tail = 1
    for i in range(2, k+1):
        tail *= i

    return head//tail
```
