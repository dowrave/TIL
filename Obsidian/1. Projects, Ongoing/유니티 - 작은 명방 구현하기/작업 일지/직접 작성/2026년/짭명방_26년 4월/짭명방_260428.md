### KeyBindingManager 구현 과정
```cs
public enum KeyAction
{
    None,
    ReturnToLobby,
    Pause,
    ChangeSpeed
}
```
이걸 구현한 다음, 각 키에 바인딩되는 값을 할당하는 개념
디폴트 설정도 자체적으로 갖는다

```cs
// 여기서 Key 타입은 빌트인이 아닌 유니티에서 제공하는 별도의 Input System임
// 이번에 설치함
using UnityEngine.InputSystem;

private Dictionary<KeyAction, Key> defaultBindings = new()
{
	{ KeyAction.ReturnToLobby, Key.Escape },
	{ KeyAction.Pause, Key.Space },
	{ KeyAction.ChangeSpeed, Key.Tab },
};

private void InitializeBindings()
{
	// enum의 각 값을 가져오는 방법에 유의(Enum.GetValues())
	foreach (KeyAction action in System.Enum.GetValues(typeof(KeyAction)))
	{
		if (action == KeyAction.None) continue;
		
		// 현재 저장 중인 키의 문자열을 가져옴, 없으면 공백
		string savedKey = PlayerPrefs.GetString(action.ToString(), "");
		
		// 값이 있으면 KeyCode로 파싱, 없으면 디폴트값 사용
		if (savedKey != "" && System.Enum.TryParse(savedKey, out Key key))
			currentBindings[action] = key;
		else
			currentBindings[action] = defaultBindings[action];
	}
}

public KeyCode GetKey(KeyAction action) => currentBindings[action];

public void SetKey(KeyAction action, Key key)
{
	currentBindings[action] = key; // 현재 바인딩 업데이트
	PlayerPrefs.SetString(action.ToString(), key.ToString()); // 설정은 string으로 저장
}

```
> 여기서 키 값을 string으로 저장하고 불러오는 방법은 익혀둘 필요는 있겠다 - 정확히는 `Enum` 타입을 순회하는 방법?

### 결과물(중간)

#### KeyBindingManager
```cs
using System.Collections.Generic;
using UnityEngine.InputSystem;
using UnityEngine;

public class KeyBindingManager : MonoBehaviour
{
    private Dictionary<KeyAction, Key> defaultBindings = new()
    {
        { KeyAction.ReturnToLobby, Key.Escape },
        { KeyAction.Pause, Key.Space },
        { KeyAction.ChangeSpeed, Key.Tab },
    };

    // 현재의 바인딩
    private Dictionary<KeyAction, Key> currentBindings = new();
    public IReadOnlyDictionary<KeyAction, Key> CurrentBindings => currentBindings;

    private void Awake()
    {
        InitializeBindings();
    }

    private void InitializeBindings()
    {
        foreach (KeyAction action in System.Enum.GetValues(typeof(KeyAction)))
        {
            if (action == KeyAction.None) continue;

            // 현재 저장 중인 키의 문자열을 가져옴, 없으면 공백
            string savedKey = PlayerPrefs.GetString(action.ToString(), "");

            // 값이 있으면 string값을 파싱, 없으면 디폴트값 사용
            if (savedKey != "" && System.Enum.TryParse(savedKey, out Key key))
                currentBindings[action] = key;
            else
                currentBindings[action] = defaultBindings[action];
        }
    }

    // 불러오기
    // --- 문자열이 필요하다면 KeyCode.ToString()을 쓰면 됨
    public Key GetKey(KeyAction action) => currentBindings[action];

    // 저장 
    public void SetKey(KeyAction action, Key key)
    {
        currentBindings[action] = key; // 현재 바인딩 업데이트
        PlayerPrefs.SetString(action.ToString(), key.ToString()); // 설정은 string으로 저장
    }
}

public enum KeyAction
{
    None,
    ReturnToLobby,
    Pause,
    ChangeSpeed
}
```

#### KeyBindingSection
```cs
using UnityEngine;
using UnityEngine.UI;
using UnityEngine.InputSystem; // 유니티에서 제공하는 별도의 인풋 시스템, 빌트인과는 다름
using System.Collections.Generic;
using TMPro;
using UnityEngine.InputSystem.Utilities;
using UnityEngine.InputSystem.Controls;

public class KeyBindingSection : MonoBehaviour
{
    [Header("Key Buttons")]
    [SerializeField] private List<BindingButtonEntry> buttonEntries;

    private BindingButtonEntry? waitingEntry = null; // 현재 입력 대기 중인 entry

    private void Awake()
    {
        foreach (BindingButtonEntry entry in buttonEntries)
        {
            KeyAction action = entry.action; // 람다 함수를 쓰므로 클로저 캡쳐가 필요
            entry.button.onClick.AddListener(() => OnBindingButtonClicked(action));
        }
    }

    private void OnEnable()
    {
        ShowCurrentButtonEntries();
    }

    private void ShowCurrentButtonEntries()
    {
        IReadOnlyDictionary<KeyAction, Key> currentBindings = GameManagement.Instance.KeyBinding.CurrentBindings;

        // struct이므로 entry는 복사본임 - 텍스트는 참조형이라 괜찮지만, entry 자체의 필드가 바뀌는 상황은 class나 for을 써야 함
        foreach (BindingButtonEntry entry in buttonEntries)
        {
            if (currentBindings.TryGetValue(entry.action, out Key key))
            {
                entry.buttonText.text = key.ToString();
            }
        }
    }

    private void OnBindingButtonClicked(KeyAction action)
    {
        BindingButtonEntry entry = buttonEntries.Find(e => e.action == action);

        waitingEntry = entry;
        entry.buttonText.text = "";

        // 버튼 입력 시 발동하는 이벤트, CallOnce로 딱 1번 실행 후 자동으로 구독을 해제
        InputSystem.onAnyButtonPress.CallOnce(ctrl => OnAnyKeyPressed(ctrl));
    }

    // 버튼 입력 대기 중 버튼이 입력됐을 때 동작
    private void OnAnyKeyPressed(InputControl ctrl)
    {
        if (waitingEntry == null) return;
        if (!(ctrl is KeyControl keyControl)) return; 

        Key pressedKey = keyControl.keyCode;
        
        // 중복 체크 로직
        IReadOnlyDictionary<KeyAction, Key> currentBindings = GameManagement.Instance.KeyBinding.CurrentBindings;
        bool isDuplicate = false;
        foreach (var binding in currentBindings)
        {
            if (binding.Value == pressedKey && binding.Key != waitingEntry.Value.action)
            {
                isDuplicate = false;
                return;
            }
        }

        if (isDuplicate)
        {
            // TODO : 중복 표시 어떻게 할지 생각
            
            // 중복일 경우 다시 입력 대기
            InputSystem.onAnyButtonPress.CallOnce(ctrl2 => OnAnyKeyPressed(ctrl2));
        }
        else
        {
            // 키 반영 및 버튼 텍스트 변경
            GameManagement.Instance.KeyBinding.SetKey(waitingEntry.Value.action, pressedKey);
            waitingEntry.Value.buttonText.text = pressedKey.ToString();
            waitingEntry = null;
        }
    }

    private void OnDestroy()
    {
        foreach (BindingButtonEntry entry in buttonEntries)
        {
            entry.button.onClick.RemoveAllListeners();
        }
    }
}

[System.Serializable]
public struct BindingButtonEntry
{
    public Button button;
    public TextMeshProUGUI buttonText;
    public KeyAction action; // PlayerPrefs에는 action.ToString()으로 전달해서 사용하면 됨  
}
```
