## 버그 / 기타 수정 사항
### 1-3 스테이지 통계 패널에서 `Sting` BGM이 들리지 않는 현상
- `AudioSource` 초기화가 제대로 안 된 듯

```cs
// 수정 전 사용 중인 AudioSource 초기화 로직
public IEnumerator SetBGMDefaultSettings()
{
	// 1. Rich BGM들 초기화
	fadeCoroutine = StartCoroutine(FadeOutBGMSource(bgmRichSource));
	yield return fadeCoroutine;

	// 2. baseBGM 활성화
	fadeCoroutine = StartCoroutine(FadeInBGMSource(bgmBaseSource));
	yield return fadeCoroutine;

	isRichBGMActive = false;
}

// 수정 후 
// 소스 상태를 bgmBaseSource를 이용하게 하는 초기화 로직.
public void ResetBGMs()
{
	bgmBaseSource.Stop(); 
	bgmBaseSource.volume = 1f;
	bgmBaseSource.loop = true;
	bgmBaseSource.clip = null;

	bgmRichSource.Stop();
	bgmRichSource.volume = 0f;
	bgmRichSource.loop = true;
	bgmRichSource.clip = null;

	isRichBGMActive = false;
}
```

### StageResultPanel - 클릭 방지 시간 공통으로 구현
- 기존엔 `Clear` 때만 `AnimateStars`이 끝난 후에 클릭 방지 패널을 집어넣었음
- 근데 Sting(짧은 음악)을 추가했기 때문에 이 음악이 끝나거나 끝나기 직전 쯤부터 클릭이 가능하도록 공통 시간을 넣어두고 로직을 외부로 뺐음
```cs
public void Initialize(int stars)
{
	this.stars = stars; 
	allOperatorStats = StatisticsManager.Instance!.GetAllOperatorStats();

	StartCoroutine(BlockingClick()); // blockDurationTime 동안 패널의 클릭을 방지
	UpdateStarRating();
	UpdateHeaders();
	CreateStatItems();
	UpdateStatContainer();
	ShowRewardItemsUI(); // stars > 0 일 때에만 동작
}

private IEnumerator BlockingClick()
{
	// Time.timeScale에 영향받지 않아야 함
	yield return new WaitForSecondsRealtime(blockDurationTime);
	blockingPanel.gameObject.SetActive(false);
}
```
