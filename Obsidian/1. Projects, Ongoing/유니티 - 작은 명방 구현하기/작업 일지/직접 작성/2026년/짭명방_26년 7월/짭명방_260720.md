```cs
using UnityEngine;

public class EnemySkillCastState: EnemyState
{
    private EnemySkillController _enemySkillController;

    private EnemyBoss _boss;
    private Operator _currentTarget;

    private EnemyBossSkill _currentSkill; //현재 스킬

    private float _remainingCastTime; // 남은 시전 시간
    private float _elapsedCastTime; // 시전 후 경과 시간
    private int _nextPhaseIndex;  // 스킬이 시전 시간 동안 여러 동작이 필요한 경우 사용


    public EnemySkillCastState(EnemyStateMachine stateMachine)
        : base(stateMachine) // 부모 클래스의 생성자를 먼저 호출한 다음 본문의 생성자를 호출하는 문법
    {
        if (_owner is EnemyBoss boss)
        {
            _boss = boss;
            _enemySkillController = boss.SkillController;
        }
    }

    public override void Enter()
    {
        base.Enter();

        // 외부에서 스킬을 쓸 수 있는지에 대한 조건을 거친 다음 Enter()로 진입함
        // TryUseSkill에서도 같은 메서드로 한 번 더 점검하고, 해당 스킬의 캐스팅이 시작되며
        // 캐스팅이 정상적으로 끝나면 OnCastComplete(), 중단되면 OnCastCancelled()가 실행된다.
        if (_enemySkillController.TryUseSkill(out EnemyBossSkill activatedSkill, out Operator mainTarget))
        {
            _currentSkill = activatedSkill;
            _currentTarget = mainTarget;
            _remainingCastTime = activatedSkill.CastTime;
            _elapsedCastTime = 0f;
            _nextPhaseIndex = 0;

            // 캐스팅 시작 알림
            _currentSkill.OnCastStart(_boss); 
        }
        else
        {
            // 방어 케이스
            _remainingCastTime = 0f;
        }
    }

    public override void Update()
    {
        if (_remainingCastTime > 0f)
        {
            _remainingCastTime -= Time.deltaTime;
            _elapsedCastTime += Time.deltaTime;

            // 시전 중 추가 동작이 있는 스킬은 자신의 시간에 맞춰서 추가 동작 실행
            float[] phaseTimings = _currentSkill.CastPhaseTimings;
            while (_nextPhaseIndex < phaseTimings.Length && _elapsedCastTime >= phaseTimings[_nextPhaseIndex])
            {
                _currentSkill.OnCastPhase(_boss, _nextPhaseIndex);
                _nextPhaseIndex++;
            }

            if (_remainingCastTime <= 0f)
            {
                _currentSkill.OnCastComplete(_boss, _currentTarget);
            }

            return; // 캐스팅 중일 때는 다른 행동 없이 대기
        }

        // 캐스팅 종료 / 실패 시 다음 상태 재판단
        if (EnemyStateTransitionCondition.CanAttack(_boss))
        {
            _fsm.ChangeState<EnemyAttackState>();
        }
        else
        {
            _fsm.ChangeState<EnemyMoveState>();
        }
    }

    public override void Exit()
    {
        // 완료되지 못한 채 상태가 빠져나가는 경우 = 중단
        if (_remainingCastTime > 0f && _currentSkill != null)
        {
            _currentSkill.OnCastCancelled(_boss);
        }

        _currentSkill = null;
        _currentTarget = null;
        _remainingCastTime = 0f;
        _elapsedCastTime = 0f;
        _nextPhaseIndex = 0;
    }
}
```

```cs
public static class EnemyStateTransitionCondition
{
    // 공격 상태 전환 조건
    public static bool CanAttack(Enemy owner)
    {
        // 저지당했을 때(이미 result에서 체크하지만 한 번 더 처리)
        if (owner.BlockingOperator != null) return true;
        
        // 공격 대상이 있고, 현재 공격이 가능한 상태일 때
        if (owner.Attack.TargetsInRange.Count > 0 && owner.Attack.AttackCooldown <= 0f) return true;

        return false;
    }

    // 스킬 상태 전환 조건 : 스킬을 쓸 수 있는가?
    public static bool CanUseSkill(Enemy _owner)
    {
        return _owner is EnemyBoss boss && 
            boss.SkillController.HasUsableSkill();
    }
}
```

#### 과정 : EnemyBossSkill - 스킬 캐스팅 메서드
- 처음에는 아래처럼 생각했음. AI에게 던졌던 질문 내용을 그대로 옮겨왔다.
```
- 기존에는 스킬 캐스팅 동작은 `yield return new WaitForSeconds()`로 주체에서 코루틴이 돌아가게끔 처리했음.
  
이제부터 구현할 메서드는 아래 로직을 따름.

1. 자신의 캐스팅 VFX나 사운드를 재생함
2. 실행 주체의 상태를 EnemySkillCastState로 바꾸고 상태에 CastTime 전달
3. 자신은 무한 대기, CastTime이 완료됐다는 신호를 받은 뒤에 실제 스킬 실행으로 넘어감
4. 한편 CastTime이 완료되지 못했을 때는 스킬 실행 없이 VFX와 사운드가 끊기고 그대로 종료됨
```

```cs
public override void Update()
```

```cs
	MoveResult result = _enemyMoveController.OnUpdate();

	if (EnemyStateTransitionCondition.CanUseSkill(_owner))

		_fsm.ChangeState<EnemySkillCastState>();
		return; 
	}

	// state에서는 result에 따라 상태 변경을 결정함

}
```
