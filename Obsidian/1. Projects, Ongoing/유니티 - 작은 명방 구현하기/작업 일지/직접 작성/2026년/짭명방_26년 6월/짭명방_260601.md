	- 영어는 `descender`가 고정이라서 이 현상이 없음

- 폰트 자체의 `Line Height`를 만지라든가, `TMP`의 `Line Height`를 만지라든가 하는 사안이 있었으나 모두 적용되지 않았음. 줄 사이의 간격을 조절하는 옵션이지 줄 자체의 높이가 변하는 옵션은 아니었던 것으로 보임.
#### 수정 완료
- 클로드 만세...
- 과정은 이렇다.
	1. **`currentStep`에 있는 전체 텍스트를 투명한 태그로 감싸서 넣음.** 
		1. 글자가 미리 있기 때문에 줄의 전체 레이아웃이 잡힌 상태로 유지된다. 
	2. 1번만 하면, **리치 텍스트의 태그가 안에 있는 경우 다른 글자는 투명한데 태그 안의 글자는 자기 색깔이 나타남.**
	3. 그래서 **최초의 전체 텍스트만 내부의 리치 텍스트 태그를 제거**함. 그러면 전체적으로 투명한 글자만 있는 상태가 되는데, 여기서 (기존 로직) 원본 전체 텍스트가 1글자씩 투명한 태그 밖으로 빠지게 되고, 나머지 글자들은 여전히 투명한 글자로 남아있는 형태가 된다.
		- 태그 처리도 사전에 해놓은 게 있어서 그걸 그대로 쓴다.

- 아래는 전체 코드다. 이번에 수정하면서 어떤 로직이 들어갔는가만 남겨둠.
```cs
// 리치 텍스트의 태그를 제거함
private string StripRichTextTags(string text)
{
	return System.Text.RegularExpressions.Regex.Replace(text, "<.*?>", string.Empty);
}

private IEnumerator TypeText()
{
	// 태그를 제거한 순수 텍스트 (레이아웃 고정 + invisiblePart 슬라이싱용)
	string strippedText = StripRichTextTags(fullText);

	// 투명한 전체 텍스트를 미리 렌더링
	// 레이아웃 관련한 이슈 : 한글 폰트는 특정 글자에 의해 줄 전체가 내려앉는 현상이 있어서 고치기 위해 이를 적용함
	textComponent.text = $"<Color=#00000000>{strippedText}</color>";

	int index = 0;
	int visibleCharCount = 0; // 실제로 출력된 보이는 문자 수(태그는 제외)

	while (index < fullText.Length)
	{
		// 태그 처리 : <로 시작해서 >로 닫히는 부분까지는 한꺼번에 추가됨
		if (fullText[index] == '<')
		{
			int tagEndIndex = fullText.IndexOf('>', index);
			if (tagEndIndex != -1)
			{
				string tag = fullText.Substring(index, tagEndIndex - index + 1);
				currentText += tag;
				index = tagEndIndex + 1;
	
				continue;
			}
		}
		
		// 일반 문자는 currentText에 추가
		currentText += fullText[index];
		index++;
		visibleCharCount++;

		// 보이는 부분(태그 포함) + 보이지 않는 나머지(태그 없는 순수 텍스트)
		string invisiblePart = strippedText.Substring(visibleCharCount);
		textComponent.text = $"{currentText}<color=#00000000>{invisiblePart}</color>";
		yield return new WaitForSecondsRealtime(typingSpeed);
	}
	
	// 완료 후 fullText 그대로 교체
	textComponent.text = fullText;
	isTyping = false;
	OnCurrentPageFinish(); // 전체 글자 등장 후의 처리
}
```

private bool _isTutorialRunning = false;
public bool IsTutorialRunning
{
	get => _isTutorialRunning;
	set
	{
		_isTutorialRunning = value;
		if (GameManagement.Instance.PlayerDataManager != null)
		{
