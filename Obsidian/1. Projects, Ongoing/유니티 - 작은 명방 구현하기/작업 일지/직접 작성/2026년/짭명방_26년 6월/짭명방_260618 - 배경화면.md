## 배경화면 구현

### 결과물(WaveLines.cs)
```cs
using UnityEngine;
using UnityEngine.UI;

[RequireComponent(typeof(CanvasRenderer))]
public class WaveLines: MaskableGraphic // MaskableGraphic : Image, Text 등의 부모 클래스. 모양 구현을 위해 필요
{
    [Header("Line Settings")]
    [SerializeField] private float lineThickness = 3f; // 선의 두께 추가

    [Header("Wave Colors (Gradient)")]
    [SerializeField] private Color topColor = new Color(0.8f, 0.5f, 1f, 1f); 
    [SerializeField] private Color bottomColor = new Color(0.2f, 0.1f, 0.8f, 0f); 

    [Header("Base Wave Settings")]
    [SerializeField] private float baseAmplitude = 50f;
    [SerializeField] private float baseFrequency = 1f;
    [SerializeField] private float baseSpeed = 2f;

    [Header("Base Wave Variation Settings")]
    [Tooltip("기준값에서 이 수치만큼 +- 범위로 무작위로 배정됩니다.")]
    [SerializeField] private float amplitudeVariation = 15f;  // 레이어마다 높낮이 차이
    [SerializeField] private float frequencyVariation = 0.5f; // 레이어마다 촘촘함 차이
    [SerializeField] private float speedVariation = 0.5f;     // 레이어마다 흐르는 속도 차이
    [SerializeField] private float phaseOffset = 2f;          // 위상은 0 ~ phaseOffsset 사이의 랜덤 값을 지정

    [Header("Second Wave Settings")]
    [Tooltip("2차 파동의 기준 비율과 각 레이어별 무작위 변동폭")]
    [SerializeField] private float secondAmplitudeBase = 0.4f;
    [SerializeField] private float secondAmplitudeVariation = 0.15f; 

    [SerializeField] private float secondFrequencyBase = 1.5f;
    [SerializeField] private float secondFrequencyVariation = 0.3f; 

    [SerializeField] private float secondSpeedBase = 1.2f;
    [SerializeField] private float secondSpeedVariation = 0.2f;

    [Header("Resolution")]
    [SerializeField] private int segmentCount = 100;

    private int waveCount = 0;

    // UI 머티리얼이 기본 텍스처(하얀색 단색)를 바탕으로 올바르게 작동하도록 보장
    public override Texture mainTexture => s_WhiteTexture;

    // 각 레이어에 할당되는 고유 무작위값을 저장하는 구조체
    private struct WaveLayerData
    {
        public float amplitude;
        public float frequency;
        public float speed;
        public float phase;

        // 2차 파동 데이터
        public float secAmplitudeRatio;
        public float secFrequencyRatio;
        public float secSpeedRatio;
    }

    private WaveLayerData[] layerDataArray;

    #if UNITY_EDITOR
    protected override void OnValidate()
    {
        base.OnValidate();
        GenerateRandomWaveData(); // 인스펙터에서 값 수정할 때마다 랜덤값을 다시 뽑아서 실시간으로 확인함
        SetVerticesDirty();
    }
    #endif

    // 동작 시점 : BackgroundManager.Start
    public void Initialize(int clearCount)
    {
        waveCount = clearCount;
        GenerateRandomWaveData();
    }

    private void Update()
    {
        if (waveCount > 0)
        {
            // 매 프레임 모양을 새로 그림
            SetVerticesDirty();
        }
    }

    // 각 레이어의 랜덤 특성을 생성하고 저장함
    private void GenerateRandomWaveData()
    {
        if (waveCount <= 0) waveCount = 0;

        layerDataArray = new WaveLayerData[waveCount];

        for (int i = 0; i < waveCount; i++)
        {
            layerDataArray[i] = new WaveLayerData
            {
                // 기준값에서 -var ~ +var 사이의 무작위 값
                amplitude = baseAmplitude + Random.Range(-amplitudeVariation, +amplitudeVariation),
                frequency = baseFrequency + Random.Range(-frequencyVariation, +frequencyVariation),
                speed = baseSpeed + Random.Range(-speedVariation, +speedVariation),
                phase = Random.Range(0f, +phaseOffset),

                // 2차 파동 설정
                secAmplitudeRatio = secondAmplitudeBase + Random.Range(-secondAmplitudeVariation, secondAmplitudeVariation),
                secFrequencyRatio = secondFrequencyBase + Random.Range(-secondFrequencyVariation, secondFrequencyVariation),
                secSpeedRatio = secondSpeedBase + Random.Range(-secondSpeedVariation, secondSpeedVariation)
            };
        }
    }

    protected override void OnPopulateMesh(VertexHelper vh)
    {
        vh.Clear();

        if (waveCount <= 0 || layerDataArray == null || layerDataArray.Length != waveCount) return;

        Rect rect = rectTransform.rect;
        float width = rect.width;
        float startX = rect.xMin;

        // 레이어 수만큼 반복해서 그리기
        for (int layer = 0; layer < waveCount; layer++)
        {
            // 캐싱해둔 레이어만의 무작위 값을 가져옴
            float currentAmplitude = layerDataArray[layer].amplitude;
            float currentFrequency = layerDataArray[layer].frequency;
            float currentSpeed = layerDataArray[layer].speed;
            float currentPhase = layerDataArray[layer].phase;
            float secAmp = currentAmplitude * layerDataArray[layer].secAmplitudeRatio;
            float secFreq = currentFrequency * layerDataArray[layer].secFrequencyRatio;
            float secSpeed = currentSpeed * layerDataArray[layer].secSpeedRatio;

            // 현재 레이어의 시작 인덱스를 기억 : 이전 레이어가 그린 다음 점에서 시작
            int currentLayerStartIndex = vh.currentVertCount;

            // 현재 레이어의 버텍스 생성
            for (int i = 0; i <= segmentCount; i++)
            {
                float t = (float) i / segmentCount;
                float x = startX + (width * t);

                // 시간과 위상은 +로 연결해줘야 함
                float firstWaveY = Mathf.Sin(t * Mathf.PI * 2f * currentFrequency + Time.time * currentSpeed + currentPhase) * currentAmplitude;
                float secondWaveY = Mathf.Sin(t * Mathf.PI * 2f * secFreq + (Time.time * secSpeed) + currentPhase) * secAmp;                
                float waveY = firstWaveY + secondWaveY;

                // 상단 점
                UIVertex vTop = UIVertex.simpleVert;
                vTop.color = topColor;
                // vTop.position = new Vector3(x, waveY, 0);
                vTop.position = new Vector3(x, waveY + (lineThickness * 0.5f), 0);
                vh.AddVert(vTop);

                // 하단 점
                UIVertex vBot = UIVertex.simpleVert;
                vBot.color = bottomColor;
                // vBot.position = new Vector3(x, -waveY, 0);
                vBot.position = new Vector3(x, waveY - (lineThickness * 0.5f), 0);
                vh.AddVert(vBot);
            }

            // 현재 레이어의 삼각형(면) 연결
            for (int i = 0; i < segmentCount; i++)
            {
                // currentLayerStartIndex를 더해서 이전 레이어의 점들과 섞이지 않게 함
                int idx = currentLayerStartIndex + (i * 2);
                vh.AddTriangle(idx, idx + 2, idx + 1);
                vh.AddTriangle(idx + 1, idx + 2, idx + 3);
            }
        }
    }
}
```



### 개요
- 옛날에 **PSP를 보면 파동 형태의 선이 은은하게 흘러가는 느낌**이 있었다. 그런 느낌으로 하나 만들어보고 싶음.
- AI에게 던져주고 스크립트만 받는 식으로 구현해도 되겠지만, 과정을 따라가는 식으로 구현함
	- 어제 블로그 개편하면서 느낀 건데 AI 딸깍에 의존해버리면 나중에 문제가 생겼을 때 어디를 고쳐야 할지 감이 아예 안 잡혀서 어떻게 질문해야 할지 모르겠더라. 

- 기본 : **컴퓨터는 선을 포함한 다른 도형을 모른다. 오직 삼각형만 알고 있다.** 
	- 2D를 포함한 개념임.

### 2. 파동
```cs
using UnityEngine;
using UnityEngine.UI;

[RequireComponent(typeof(CanvasRenderer))]
public class WaveLine: MaskableGraphic // MaskableGraphic : Image, Text 등의 부모 클래스. 모양 구현을 위해 필요
{
    public float thickness = 5f; // 선의 두께
    public float amplitude = 30f; // 파동의 높이
    public float frequency = 0.01f; // 파동의 촘촘함
    public float speed = 5f; // 파동이 흘러가는 속도
    public int segmentCount = 100; // 선을 쪼개는 정도(해상도)    

    private void Update()
    {
        // 매 프레임 모양을 새로 그림
        SetVerticesDirty();
    }

    protected override void OnPopulateMesh(VertexHelper vh)
    {
        vh.Clear();

        // RectTransform의 크기를 가져와 화면 좌우 꽉 채움
        Rect rect = rectTransform.rect;
        float width = rect.width;
        float startX = rect.xMin; // 시작점 : UI 영역의 가장 왼쪽 X 좌표

        float halfThickness = thickness / 2f;

        // x축 기준으로 점을 위아래로 2개씩 찍음
        for (int i = 0; i <= segmentCount; i++)
        {
            // 0 ~ 1 사이의 진행률
            float t = (float) i / segmentCount;

            // 현재 세로줄의 x좌표
            float x = startX + (width * t);

            // 현재 x 좌표와 시간에 기반한 Y좌표(=파동의 높이) 계산
            float waveY = Mathf.Sin((x * frequency) + (Time.time * speed)) * amplitude;

            // 점(버텍스) 정의
            UIVertex v = UIVertex.simpleVert;
            v.color = color;

            // 위쪽 점 생성 (짝수 인덱스)
            v.position = new Vector3(x, waveY + halfThickness, 0);
            vh.AddVert(v);

            // 아랫쪽 점 생성 (홀수 인덱스)
            v.position = new Vector3(x, waveY - halfThickness, 0);
            vh.AddVert(v);
        }

        // 생성된 점들을 이어서 직사각형 만들기
        for (int i = 0; i < segmentCount; i++) 
        {
            // i번 구간의 직사각형을 만들기 위해 4개의 점 인덱스 계산
            int index = i * 2;

            // index : 현재 기둥의 윗쪽 점 (좌상)
            // index + 1 : 현재 기둥의 아랫쪽 점(좌하)
            // index + 2 : 다음 기둥의 윗쪽 점(우상)
            // index + 3 : 다음 기둥의 아랫쪽 점(우하)

            vh.AddTriangle(index, index + 2, index + 1); // 좌상 - 우상 - 좌하
            vh.AddTriangle(index + 1, index + 2, index + 3); // 우상 - 우하 - 좌하
        }
    }
}
```

### 3. PSP 배경화면 느낌 내기
![[Pasted image 20260618173634.png]]
> 요런 느낌을 원함

- 대충 이런 아이디어들이 반영됨
1. 위 파동에선 "선"을 만들었지만, 이런 구현은 **화면의 바닥 ~ 파동까지 이어진 면**으로 접근
2. 부드럽게 밀려가는 효과를 위해, **`Mathf.Sin()` 1개가 아닌 여러 삼각함수를 합친 형태**
3. `frequency`의 경우 `Mathf.PI` 없이는 인스펙터에서 조절할 때 이상하게 깨지는 경우가 잦음 \
4. 그라데이션의 조절을 위해, x좌표 1개당 점을 3개씩 찍음, 중간 점의 위치는 `ratio`로 조절 할 수 있게 구현

- 일단은 PSP 배경화면까지 미치진 못하더라도, 이런 느낌으로 구현함. 근데 PSP 배경화면 느낌도 충분히 구현할 수 있을 듯.

```cs
using UnityEngine;
using UnityEngine.UI;

[RequireComponent(typeof(CanvasRenderer))]
public class WaveLine: MaskableGraphic // MaskableGraphic : Image, Text 등의 부모 클래스. 모양 구현을 위해 필요
{
    [Header("Wave Colors (Gradient)")]
    // [ColorUsage(true, true)]
    public Color topColor = new Color(0.8f, 0.5f, 1f, 1f); 
    // [ColorUsage(true, true)]
    public Color middleColor = new Color(0.8f, 0.5f, 1f, 1f); 
    // [ColorUsage(true, true)]
    public Color bottomColor = new Color(0.2f, 0.1f, 0.8f, 0f); 

    [Header("Gradient Control")]
    [Range(0f, 1f)]
    [Tooltip("0이면 바닥, 1이면 꼭대기. 값이 클수록 투명한 영역이 위로 올라옵니다.")]
    public float midPointRatio = 0.6f; // 중간점의 위치 비율

    [Header("Primary Wave (크고 느린 파도)")]
    public float amplitude1 = 50f;
    public float frequency1 = 0.005f;
    public float speed1 = 2f;

    [Header("Secondary Wave (작고 불규칙함을 만드는 파도)")]
    public float amplitude2 = 20f;
    public float frequency2 = 0.015f;
    public float speed2 = 3.5f;

    [Header("Resolution")]
    public int segmentCount = 100;

    private void Update()
    {
        // 매 프레임 모양을 새로 그림
        SetVerticesDirty();
    }

    protected override void OnPopulateMesh(VertexHelper vh)
    {
        vh.Clear();

        Rect rect = rectTransform.rect;
        float width = rect.width;
        float startX = rect.xMin;
        float bottomY = rect.yMin; // 파동의 아랫부분을 화면의 가장 밑바닥으로 고정

        for (int i = 0; i <= segmentCount; i++)
        {
            float t = (float)i / segmentCount;
            float x = startX + (width * t);

            // 2개의 서로 다른 사인파를 더해 불규칙하고 자연스러운 곡선 생성
            float waveY = (Mathf.Sin(t * Mathf.PI * 2f * frequency1 + Time.time * speed1) * amplitude1) 
                        + (Mathf.Sin(t * Mathf.PI * 2f * frequency2 + Time.time * speed2) * amplitude2);

            float midY = Mathf.Lerp(bottomY, waveY, midPointRatio);

            // 상단 점 (밝고 불투명한 색상)
            UIVertex vTop = UIVertex.simpleVert;
            vTop.color = topColor;
            vTop.position = new Vector3(x, waveY, 0);
            vh.AddVert(vTop);

            // ② 중간 점 (투명도를 끌어올리는 기준점 / 투명한 색상)
            UIVertex vMid = UIVertex.simpleVert;
            vMid.color = middleColor; 
            vMid.position = new Vector3(x, midY, 0);
            vh.AddVert(vMid);

            // 하단 점 (어둡거나 투명한 색상)
            UIVertex vBottom = UIVertex.simpleVert;
            vBottom.color = bottomColor;
            vBottom.position = new Vector3(x, bottomY, 0);
            vh.AddVert(vBottom);
        }

        // 삼각형(면) 연결
        for (int i = 0; i < segmentCount; i++)
        {
            // 한 세로줄 당 점이 3개씩 들어가므로 인덱스는 3의 배수로 증가
            int idx = i * 3; 

            // 상단 직사각형 그리기 (Top ~ Mid 구간)
            vh.AddTriangle(idx, idx + 3, idx + 1);
            vh.AddTriangle(idx + 1, idx + 3, idx + 4);

            // 하단 직사각형 그리기 (Mid ~ Bot 구간)
            vh.AddTriangle(idx + 1, idx + 4, idx + 2);
            vh.AddTriangle(idx + 2, idx + 4, idx + 5);
        }
    }
}    
```

### 4. 여러 개의 선 구현하기
- **클리어 스테이지 수에 맞춰서 배경 화면이 바뀌는 효과**를 내겠다면 화면의 최하단에 버텍스를 두는 3개의 버텍스를 쓰는 방식보다는 **2개의 버텍스**를 쓰는 방식이 나아보임
	- 대신 두 버텍스 사이에는 그라데이션 효과를 구현
- 각 파동은 랜덤한 바리에이션을 갖도록 구현
	- 진폭, 진동수, 속도
	- `OnPopulateMesh()` 내부에서 `Random.Range`를 직접 사용하면 지터 현상이 발생한다고 한다. 따라서 **레이어별 랜덤값을 미리 뽑아서 캐싱한 다음 이를 바탕으로 그려야 함**

- 구현했는데, 코드는 아래에서 한꺼번에 적어놓음

### 기타 이슈 : 머티리얼이 적용되지 않음
- 여기에 추가로 Additive 머티리얼을 적용해서 선이 겹칠 때 더 밝아보이게끔 구현하려고 했는데, 선이 진한 붉은색으로 나타나는 이슈가 있다. 이거 해결하는 중.
	- **`Screen Space - Overlay` 캔버스 내에 `URP - Unlit` 셰이더가 들어가면, `Vertex Color` 데이터를 넘기는 과정이 실패하는 경우가 있다**고 함
		- 정상적으로 `Vertex Color`를 받지 못한 셰이더는 에러값으로 기본값인 `(1, 0, 0, 1)`인 붉은색을 출력하게 되는데, 이게 `Additive` 블렌딩과 섞여서 진한 붉은새으로 보이게 된다는 것.
-  해결) **`Screen Space - Camera`에 해당하는 캔버스를 하나 더 생성, 해당 카메라는 `Main Camera`로 지정**


### 클리어 갯수와 연결된 최종본
- 스테이지 클리어 갯수를 가져와서, 배경화면에 선을 그리는 로직까지 작성했음

#### `OnPopluateMesh()`의 동작에 관해
- 기존에는 `Awake`에서 바로 실행했는데, 지금처럼 매니저의 `Start`에서 `WaveLines.Initialize(int clearCount)`가 동작했을 때 어떻게 동작하는지 궁금함
- 그리는 로직은 모두 **`OnPopulateMesh()`** 에 들어가 있다. **이 메서드는 유니티의 UI 렌더링 시스템이 다시 그려야한다고 판단했을 때 자동으로 호출하는 콜백 함수**다.

- 과정
1. `waveCount` 값이 `Initialize`에 의해 0 이상으로 설정되면, `Update()`에서 `SetVerticesDirty()`가 호출한다.
	- **`SetVerticesDirty()`는** 당장 메쉬를 그리는 명령이 아니고, 모양이 바뀌었음을 알려서 **렌더링 전에 다시 그려달라고 예약을 거는 함수**임
2. 유니티가 화면을 그리기 직전, `Dirty` 플래그가 꽂힌 UI 요소들을 모아서 `OnPopulateMesh`를 호출한다. 이 때 화면에 그려진다.

즉, 값이 들어간 다음에 `OnPopulateMesh`가 동작하게 되고, 이게 동작하는 플래그를 `SetVerticesDirty`가 설정한다는 의미.

#### 캔버스 분리
- 머티리얼이 `Screen Space - Overlay`에 적용되지 않는 상황 떄문에, 자연스럽게 캔버스를 분리하게 되었다.

```
현재의 캔버스 설정
- backgroundCanvas(waveLines)
- Canvas(UI 요소들)
```

- 만약 **`waveLines`가 기존 캔버스에 있었다면, `SetVerticesDirty()`에 의해 다른 UI 요소들도 모두 다시 구워지는 필요없는 동작이 발생**했을 것이다.
- 캔버스를 분리했기 때문에, `backgroundCanvas`만 계속해서 새로고침된다. 
	- 이런 식으로 **동적인 UI와 정적인 UI를 별도의 캔버스에 그리는 건 유니티 매뉴얼에서도 권장하는 최적화 기법**이다. 

#### 연산량
- 메쉬를 새로 만들고 있는 현재 상황.
- 세그먼트가 100개이고 클리어 횟수가 5개라면, 매 프레임 C#은 약 500번의 `Mathf.Sin` 연산을 수행하고 1000개의 버텍스를 새로 찍어내게 된다. 
	- PC, 최신 모바일 기기 기준 아주 가벼운 연산량이므로 전혀 문제되지 않는다. 저사양 기기에 적용할 경우에나 최적화가 필요한데, 이건 지금 고려하지 않아도 됨.
