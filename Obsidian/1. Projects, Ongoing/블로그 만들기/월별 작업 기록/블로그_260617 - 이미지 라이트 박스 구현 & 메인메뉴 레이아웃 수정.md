## 이미지 라이트박스
- 이름이 `Lightbox`인 이유 
	- 웹에서 오래 전부터 쓰인 UI 패턴의 이름이다.
	- 사진 촬영에서 필름 / 슬라이드 등을 뒤에서 비춰 선명하게 보는 장치를 라이트박스라고 했는데, **웹에서는 배경을 어둡게 깔고 컨텐츠(주로 이미지)를 중앙에 크게 띄워 집중시키는 UI를 라이트박스**라고 통칭하게 됐다. 
### ImageLightBox.tsx - UI 컴포넌트
```tsx
import React, { useState, useEffect, useCallback, useRef } from 'react';
import ReactDOM from 'react-dom';

import { IoClose } from 'react-icons/io5';
import { FiZoomIn, FiZoomOut } from 'react-icons/fi';
import { MdOutlineZoomOutMap } from 'react-icons/md';

interface ImageLightboxProps {
  src: string;
  alt?: string;
  onClose: () => void;
}

const isMobile = () => window.matchMedia('(pointer: coarse)').matches;

const MIN_SCALE = 0.5; // 최소 배율 (최대한 작게 보임)
const MAX_SCALE = 3; // 최대 배율 (세세한 부분까지 보임)
const SCALE_STEP = 0.25;

// 이미지 축소 / 확대를 위한 컴포넌트. 
const ImageLightbox: React.FC<ImageLightboxProps> = ({ src, alt = '이미지', onClose }) => {
  const [scale, setScale] = useState(1);
  const [translate, setTranslate] = useState({ x: 0, y: 0 });

  // 드래그 상태
  const isDraggingRef = useRef(false);
  const lastPointerRef = useRef({ x: 0, y: 0 });
  const overlayRef = useRef<HTMLDivElement>(null); // 오버레이 DOM 요소 참조를 위한 ref

  // 핀치-줌 상태 (모바일)
  const lastPinchDistRef = useRef<number | null>(null);

  // ── 유틸 ──────────────────────────────────────────────
  const clampScale = (s: number) => Math.min(MAX_SCALE, Math.max(MIN_SCALE, s));

  const resetView = useCallback(() => {
    setScale(1);
    setTranslate({ x: 0, y: 0 });
  }, []);

  // ── 키보드 ────────────────────────────────────────────
  useEffect(() => {
    const onKey = (e: KeyboardEvent) => {
      if (e.key === 'Escape') onClose();
      if (e.key === '+' || e.key === '=') setScale(s => clampScale(s + SCALE_STEP));
      if (e.key === '-') setScale(s => clampScale(s - SCALE_STEP));
      if (e.key === '0') resetView();
    };
    window.addEventListener('keydown', onKey);
    return () => window.removeEventListener('keydown', onKey);
  }, [onClose, resetView]);

  // ── 스크롤 방지 ───────────────────────────────────────
  useEffect(() => {
    const prev = document.body.style.overflow;
    document.body.style.overflow = 'hidden';
    return () => { document.body.style.overflow = prev; };
  }, []);

  // ── 마우스 휠 줌 ─────────────────────────────────────
  useEffect(() => {
    const el = overlayRef.current;
    if (!el) return;
    const onWheel = (e: WheelEvent) => {
        e.preventDefault();
        e.stopPropagation();
        
        // 아래 스크롤 시 이미지가 작아짐
        const delta = e.deltaY < 0 ? SCALE_STEP : -SCALE_STEP;

        setScale(s => clampScale(s + delta));
    };
    
    // 기본 설정인 passive: true으로 등록 시 preventDefault()가 무시된다.
    // 네이티브 이벤트로 등록하면서 passive: false를 명시해서 preventDefault()와 stopPropagation()이 실제로 동작하게 함
    el.addEventListener('wheel', onWheel, { passive: false });

    return () => el.removeEventListener('wheel', onWheel);
  })


// 이벤트 전파가 뚫리는 경우가 있어서 상단의 useEffect()로 수정
//   const handleWheel = useCallback((e: React.WheelEvent) => {
//     e.preventDefault();
//     const delta = e.deltaY < 0 ? SCALE_STEP : -SCALE_STEP;
//     setScale(s => clampScale(s + delta));
//   }, []);

  // ── 마우스 드래그 ─────────────────────────────────────
  const handleMouseDown = useCallback((e: React.MouseEvent) => {
    if (scale <= 1) return; // 원본 크기일 땐 드래그 비활성
    isDraggingRef.current = true;
    lastPointerRef.current = { x: e.clientX, y: e.clientY };
    e.preventDefault();
  }, [scale]);

  const handleMouseMove = useCallback((e: React.MouseEvent) => {
    if (!isDraggingRef.current) return;
    const dx = e.clientX - lastPointerRef.current.x;
    const dy = e.clientY - lastPointerRef.current.y;
    lastPointerRef.current = { x: e.clientX, y: e.clientY };
    setTranslate(t => ({ x: t.x + dx, y: t.y + dy }));
  }, []);

  const stopDrag = useCallback(() => { isDraggingRef.current = false; }, []);

  // ── 터치 (드래그 + 핀치줌) ───────────────────────────
  const handleTouchStart = useCallback((e: React.TouchEvent) => {
    if (e.touches.length === 2) {
      const dx = e.touches[0].clientX - e.touches[1].clientX;
      const dy = e.touches[0].clientY - e.touches[1].clientY;
      lastPinchDistRef.current = Math.hypot(dx, dy);
    } else if (e.touches.length === 1) {
      lastPointerRef.current = { x: e.touches[0].clientX, y: e.touches[0].clientY };
      isDraggingRef.current = true;
    }
  }, []);

  const handleTouchMove = useCallback((e: React.TouchEvent) => {
    e.preventDefault();
    if (e.touches.length === 2 && lastPinchDistRef.current !== null) {
      const dx = e.touches[0].clientX - e.touches[1].clientX;
      const dy = e.touches[0].clientY - e.touches[1].clientY;
      const dist = Math.hypot(dx, dy);
      const ratio = dist / lastPinchDistRef.current;
      lastPinchDistRef.current = dist;
      setScale(s => clampScale(s * ratio));
    } else if (e.touches.length === 1 && isDraggingRef.current && scale > 1) {
      const dx = e.touches[0].clientX - lastPointerRef.current.x;
      const dy = e.touches[0].clientY - lastPointerRef.current.y;
      lastPointerRef.current = { x: e.touches[0].clientX, y: e.touches[0].clientY };
      setTranslate(t => ({ x: t.x + dx, y: t.y + dy }));
    }
  }, [scale]);

  const handleTouchEnd = useCallback(() => {
    isDraggingRef.current = false;
    lastPinchDistRef.current = null;
  }, []);

  // ── 스케일 리셋 (1배 이하로 내리면 translate도 초기화) ──
  useEffect(() => {
    if (scale <= 1) setTranslate({ x: 0, y: 0 });
  }, [scale]);

  // ── 렌더 ─────────────────────────────────────────────
const portal = (
    <div
      ref = {overlayRef}
      className="fixed inset-0 z-[9999] flex items-center justify-center bg-black/85 backdrop-blur-sm"
      onClick={onClose}
      style={{ touchAction: 'none' }}
    >
        
      {/* 컨트롤 버튼 (우측 상단) — 데스크탑 전용 */}
      <div
        className="absolute top-3 right-3 flex items-center gap-2 z-10"
        onClick={e => e.stopPropagation()}
      >
        {!isMobile() && (
          <>
            {/* 여기 div들은 원래 Button이었으나 flex items-center justify-center와 충돌해서 아이콘이 사라지는 이슈가 있었다.
            button 태그들을 div로 수정하니까 동작하므로 이렇게 냅둠. */}
            <div
              role = "button"
              onClick={() => setScale(s => clampScale(s - SCALE_STEP))}
              className="w-9 h-9 flex items-center justify-center rounded-full  bg-white/15 hover:bg-white/30 text-white transition-colors"
              aria-label="축소"
            >
              <FiZoomOut size={18}/>
            </div>
 
            <span className="text-white/80 text-xs font-mono w-10 text-center select-none">
              {Math.round(scale * 100)}%
            </span>
 
            <div
              role = "button"
              onClick={() => setScale(s => clampScale(s + SCALE_STEP))}
              className="w-9 h-9 flex items-center justify-center rounded-full bg-white/15 hover:bg-white/30 text-white transition-colors"
            //   className="w-9 h-9 flex items-center justify-center rounded-full text-white transition-colors"
              aria-label="확대"
            >
              <FiZoomIn size={18}/>
            </div>
 
            <div
              role = "button"
              onClick={resetView}
              className="w-9 h-9 flex items-center justify-center rounded-full bg-white/15 hover:bg-white/30 text-white transition-colors"
            //   className="w-9 h-9 flex items-center justify-center rounded-full text-white transition-colors"
              aria-label="원래 크기"
            >
              <MdOutlineZoomOutMap size={18}/>
            </div>
          </>
        )}
 
        {/* 닫기 버튼은 모바일/데스크탑 공통 */}
        <div
          role = "button"
          onClick={onClose}
          className="w-9 h-9 flex items-center justify-center rounded-full bg-white/15 hover:bg-white/30 text-white transition-colors ml-1"
        //   className="w-9 h-9 flex items-center justify-center rounded-full  text-white transition-colors ml-1"

          aria-label="닫기"
        >
          <IoClose size={20}/>
        </div>
      </div>
 
      {/* 이미지 래퍼 */}
      <div
        className="relative max-w-[90vw] max-h-[90vh] flex items-center justify-center"
        onClick={e => e.stopPropagation()}
        // onWheel={handleWheel}
        onMouseDown={handleMouseDown}
        onMouseMove={handleMouseMove}
        onMouseUp={stopDrag}
        onMouseLeave={stopDrag}
        onTouchStart={handleTouchStart}
        onTouchMove={handleTouchMove}
        onTouchEnd={handleTouchEnd}
        style={{
          cursor: scale > 1 ? (isDraggingRef.current ? 'grabbing' : 'grab') : 'default',
          userSelect: 'none',
        }}
      >
        <img
          src={src}
          alt={alt}
          draggable={false}
          style={{
            maxWidth: '90vw',
            maxHeight: '90vh',
            objectFit: 'contain',
            transform: `scale(${scale}) translate(${translate.x / scale}px, ${translate.y / scale}px)`,
            transition: isDraggingRef.current ? 'none' : 'transform 0.15s ease',
            borderRadius: '4px',
            boxShadow: '0 8px 40px rgba(0,0,0,0.6)',
          }}
        />
      </div>
 
      {/* 하단 힌트 — 데스크탑 전용 */}
      {!isMobile() && (
        <p className="absolute bottom-4 left-0 right-0 text-center text-white/40 text-xs select-none pointer-events-none">
          스크롤로 확대/축소{scale > 1 ? ' · 드래그로 이동' : ''} · ESC로 닫기
        </p>
      )}
    </div>
  );

  return ReactDOM.createPortal(portal, document.getElementById('modal-root')!);
};

export default ImageLightbox;
```

### useImageLightBox.ts
```ts
import { useState, useEffect, useCallback } from 'react';

interface LightboxState {
  src: string;
  alt: string;
}

export function useImageLightbox(containerSelector: string) {
  const [lightbox, setLightbox] = useState<LightboxState | null>(null);

  const openLightbox = useCallback((src: string, alt: string) => {
    setLightbox({ src, alt });
  }, []);

  const closeLightbox = useCallback(() => {
    setLightbox(null);
  }, []);

  useEffect(() => {
    const handler = (e: MouseEvent) => {
      const target = e.target as HTMLElement;
      if (target.tagName !== 'IMG') return;
      if (target.closest('.ql-toolbar') || target.closest('.ql-resizer')) return;

      const img = target as HTMLImageElement;
      openLightbox(img.src, img.alt || '이미지');
    };

    // 등록된 컨테이너를 추적 (클린업에서 제거하기 위해)
    const attachedContainers: HTMLElement[] = [];

    const attach = () => {
      document
        .querySelectorAll<HTMLElement>(containerSelector)
        .forEach(c => {
          c.addEventListener('click', handler);
          attachedContainers.push(c);
        });
    };

    attach();

    // 아직 DOM에 없으면 MutationObserver로 대기
    let observer: MutationObserver | null = null;
    if (attachedContainers.length === 0) {
      observer = new MutationObserver(() => {
        const found = document.querySelectorAll<HTMLElement>(containerSelector);
        if (found.length > 0) {
          found.forEach(c => {
            c.addEventListener('click', handler);
            attachedContainers.push(c);
          });
          observer?.disconnect();
          observer = null;
        }
      });
      observer.observe(document.body, { childList: true, subtree: true });
    }

    return () => {
      attachedContainers.forEach(c => c.removeEventListener('click', handler));
      observer?.disconnect();
    };
  }, [containerSelector, openLightbox]);

  return { lightbox, closeLightbox };
}
```
- 설명(AI 요약)
	- CSS 선택자를 받아서 해당 영역 안의 이미지 클릭을 감지하고 라이트박스 열림/닫힘 상태를 관리합니다. 
	- 이벤트 위임 방식으로 동작하며, 비동기 렌더링으로 DOM이 늦게 생기는 경우를 대비해 `MutationObserver`로 컨테이너가 나타날 때까지 대기합니다.

### 진행 중 수정한 사항
#### 1. 모바일 환경에서 확대 / 축소 / 원본 크기로 보는 버튼 및 하단 힌트 박스 보이지 않게 변경

#### 2. 버튼들의 이미지 아이콘 보이지 않는 문제 수정
- `button`에 `flex items-center justify-center`을 함께 쓰면 중심에 들어가야 할 아이콘이 어디론가 벗어나 있다. 
- **`div`로 태그 변경**하면 잘 됨.
	- 버튼의 배경을 없애는 편이 깔끔하긴 한데, 배경 색이 없을 경우 이미지에 버튼이 묻혀서 보이지 않는 현상이 있다. 반투명 배경은 안전망 역할을 한다.

#### 3. `ImageLightbox`가 나타난 상태에서 마우스 스크롤이 외부까지 영향을 미치는 현상 수정

- `handleWheel`에서 `e.preventDefault()`을 사용하고 있음 
- `React`의 합성 이벤트는 `passive Listener`로 등록, `preventDefault()`가 무시되는 브라우저가 있다. 이 때만 전파가 뚫린다.
	- (스크롤이 외부까지 영향을 미치는 현상이 항상 동작하는 건 아니었음)

### 2. 뒤로 가기 클릭 시 라이트박스가 제거되는 게 아니라 링크가 이동함
- `ReviewModal`에서 썼던 `useModalHistory`를 여기에서도 사용함. 
- `useModalHistory`는 `useBackHistory`로 이름 변경.

```tsx
import { useEffect, useRef } from 'react';

// 뒤로 가기 버튼을 클릭했을 때 링크의 이동이 아니라 새로 등장한 컴포넌트만을 제거시키기 위한 훅
// 컴포넌트 예시 : reviewModal, imageLightbox
// 원리 : 컴포넌트 마운트 시 히스토리 스택에 상태를 하나 쌓고, 뒤로 가기 발생 시 onClose를 호출함
export const useBackHistory = (isOpen: boolean, onClose:() => void) => {

    // 직접 history state를 추가했는지 여부룰 추적하는 ref
    const isStatePushed = useRef(false);

    useEffect(() => {
        // 모달이 열렸을 때만 로직 실행
        if (isOpen) {

            // historyState가 없으면 추가한다. 불필요한 pushState 호출을 막는다.
            if (!isStatePushed.current) {
                window.history.pushState({ isOpen: true }, '');
                isStatePushed.current = true;
            }

            // popstate 이벤트 핸들러 정의. 사용자가 뒤로 가기를 했다는 의미.
            const handlePopState = (event: PopStateEvent) => {
                // 이 컴포넌트가 쌓은 state가 남아있다면 뒤로 가기가 아직 완료되지 않은 것이므로 무시한다.
                if (event.state?.isOpen) {
                    return;
                }

                // 진짜 뒤로가기가 실행된 상황
                onClose();
            }
            
            window.addEventListener('popstate', handlePopState);

            return () => {
                // 클린업 함수 : 컴포넌트가 언마운트되거나 isOpen = false일 때 실행
                window.removeEventListener('popstate', handlePopState);
            }
        }
        // -- 요소가 닫혔을 때의 로직 --
        else {
            if (isStatePushed.current) { 
                isStatePushed.current = false; // 상태를 먼저 변경해 무한 루프를 방지한다.
                window.history.back(); // popstate 이벤트 발생
            }
        }
    }, [isOpen, onClose])
}
```

### 3. ReviewModal 안에서 ImageLightBox를 띄웠을 때, 뒤로 가기 버튼이 각각에 대해 동작하지 않고 2번 클릭했을 때 두 동작이 한꺼번에 수행됨
- 원인
	- 두 컴포넌트의 `pushState`가 동시에 일어나기 때문이다.
	- `ReviewModal`이 열릴 때 1회, `ImageLightbox`가 열릴 때 1회로 총 2회 쌓인다.
	- 뒤로 가기를 1번 누르면 `ImageLightbox`의 `state`만 `pop`되는데, 두 컴포넌트의 `popstate` 핸들러가 동시에 실행되면서 서로 충돌하여서 1번째에는 둘 다 닫히지 않고, 2번째의 남은 `state`는 `pop` 되면서 둘 다 닫히게 된다.
- 해결법
	- **`ImageLightbox`가 열린 동안, `ReviewModal`의 `popstate` 핸들러를 무시시킨다.**
	- 상위 컴포넌트가 `ImageLightbox`라고 볼 수 있는 상황이므로, 여기서 처리함
```tsx
// ReviewModal.tsx
// 뒤로 가기 클릭 시 이 모달만 닫히게 설정. 
// lightbox가 열려 있다면 여기서의 useBackHistory는 동작하지 않음
useBackHistory(!!reviewId && !lightbox , isEditing ? handleAttemptClose : onClose);
```
> `lightBox`가 꺼져 있을 때만 `useBackHistory` 동작이 수행되게 된다.

- 답변 : `pushState` 대신 `replaceState`를 쓰자
	- `pushState`는 히스토리 스택에 항목을 추가하는 개념인데, `ImageLightbox`가 닫힐 때 `history.back()`을 호출하면 기록이 "앞으로 가기" 스택에 남게 된다.
	- **열릴 때 `pushState`로 현재 상태를 표시**하고, 
	- **닫힐 때 `history.back()`대신 `replaceState()`로 현재 항목을 덮어쓰는 방식**을 쓰면 닫힌 기록이 스택에 남지 않는다.

- 계속 
1. 여전히 앞으로 가기 등에 의한 이슈는 남아 있음
	- 앞으로 가기의 경우, 자바스크립트로 히스토리 **스택을 없애는 건 브라우저 보안 정책 상 불가능**함.
	- 그러면 스택에 해당 라이트박스나 모달이 표시되는 기능까지도 생각해봤지만, 이 기능들을 URL 기반으로 구현한 게 아니기 때문에 일단 **보류**(라이트박스, 모달 모두 별도의 URL을 사용하고 있지 않음)

2. `reviewModal`에서의 라이트박스는 뒤로 가기 1회 클릭 시 반응이 없음
	- 뒤로 가기 히스토리가 2개 생기는데, 1번째 히스토리로 가도 반응이 없고, 2번째 히스토리는 라이트박스가 꺼짐. 재밌는건 2번째 히스토리로 갔을 때 다시 1개의 히스토리가 더 생기며, 이 히스토리는 모달에 관한 히스토리.

```tsx
import { useEffect, useRef } from 'react';

export const useBackHistory = (isOpen: boolean, onClose:() => void) => {

    const isStatePushed = useRef(false);
    const instanceId = useRef(`history_${Math.random().toString(36).substring(2, 11)}`)

    useEffect(() => {
        if (isOpen) {
            if (!isStatePushed.current) {
                window.history.pushState({ historyId: instanceId.current }, '');
                isStatePushed.current = true;

                if (event.state?.historyId === instanceId.current ) return;
                onClose();

            return () => window.removeEventListener('popstate', handlePopState);
```

코드 자체는 상단에 주석처리한 걸 풀어서 적용했지만, **아직 끝나지 않았따.** 뒤로 가기는 이제 원하는 대로 잘 동작하는데, 그 외의 **정상적인 방법으로 끌 때 히스토리가 여전히 남아 있음**

컴포넌트가 아예 언마운트되기 때문으로, 찌꺼기를 제거하는 역할을 `else`문에서 `useEffect` 내부로 옮겨야 한다. 단, 이 때 `history.back()`의 경우 뒤로 가기 때에도 똑같이 동작하므로 이들을 구분하기 위한 `ref`를 하나 추가해야 한다.

- `isClosedByPopState` 부분만 적어둠
```tsx
const isClosedByPopState = useRef(false);

useEffect(() => {
	if (isOpen) {
		if (!isStatePushed.current) {
			// ... 
			isClosedByPopState.current = false;
		}
		
		const handlePopState = (event: PopStateEvent) => {
		
			isClosedByPopState.current = true; // 뒤로 가기 플래그 기록
		}
		
		// ...
		
            // 클린업 함수 : 컴포넌트 언마운트, isOpen = false가 될 때 모두 동작함
            return () => { 
                window.removeEventListener('popstate', handlePopState);

                // 히스토리를 심어둔 상태
                if (isStatePushed.current) {
                    isStatePushed.current = false;

                    // UI 요소로 닫혔을 때의 처리(뒤로 가기 X)
                    if (!isClosedByPopState.current) {
                        window.history.back();
                    }
                }
            };
        }
    }, [isOpen, onClose])
```
